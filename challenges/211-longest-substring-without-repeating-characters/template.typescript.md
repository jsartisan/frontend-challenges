    expect(lengthOfLongestSubstring("pwwkew")).toBe(3);
  });

  test('Empty string', () => {
    expect(lengthOfLongestSubstring("")).toBe(0);
  });

  test('Single character', () => {

  });

    expect(lengthOfLongestSubstring("ab c!d")).toBe(5);
